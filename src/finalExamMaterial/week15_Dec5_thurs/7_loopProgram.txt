def diff(big: Z, small: Z): Z = {
    Contract(
        Requires( big >= small ),
        Ensures ( Res[Z] == big-small )
    )

    //PREMISES:
    big >= small


    var count: Z = 0
    var cur: Z = big

    //PREMISES:
    count == 0
    cur == big
    big >= small


    //SHOW:
    count == (big-cur)


    while (cur != small) {
        Invariant(
            Modifies(count, cur),
            count == (big-cur)
        )

        //PREMISES:
        cur != small
        count == (big-cur)
        big >= small


        cur = cur - 1
        count = count + 1

        //SHOW:
        count == (big-cur)

    }

    //PREMISES:
    !(cur != small)
    count == (big-cur)
    big >= small


    //SHOW:
    count == big-small


    return count
}
